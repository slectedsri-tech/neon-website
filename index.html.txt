    <script>
        const music = document.getElementById('bgMusic');
        const btn = document.querySelector('.btn');
        
        function toggleMusic() {
            if (music.paused) {
                music.play();
                btn.innerHTML = "Pause Music";
                btn.style.backgroundColor = "#ff4757"; // Change color to red when playing
                btn.style.boxShadow = "0 4px 15px rgba(255, 71, 87, 0.4)";
            } else {
                music.pause();
                btn.innerHTML = "Toggle Music";
                btn.style.backgroundColor = "#00ff41"; // Back to green when paused
                btn.style.boxShadow = "0 4px 15px rgba(0, 255, 65, 0.4)";
            }
        }
